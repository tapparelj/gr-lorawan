257 // Table support
258 // These macros for defining a table of constants and retrieving values
259 // from it makes it easier for other platforms (like AVR) to optimize
260 // table accesses.
261 // Use CONST_TABLE() whenever declaring or defining a table, and
262 // TABLE_GET_xx whenever accessing its values. The actual name of the
263 // declared variable will be modified to prevent accidental direct
264 // access. The accessor macros forward to an inline function to allow
265 // proper type checking of the array element type.
266 
267 // Helper to add a prefix to the table name
268 #define RESOLVE_TABLE(table) constant_table_ ## table
269 
270 // get number of entries in table

272

285  // Macro to define the getter functions. This loads data from
286  // progmem using pgm_read_xx, or accesses memory directly when the
287  // index is a constant so gcc can optimize it away;
288  #define TABLE_GETTER(postfix, type, pgm_type) \
289  static inline type table_get ## postfix(const type *table, size_t index) { \
290  if (__builtin_constant_p(table[index])) \
291  return table[index]; \
292  return pgm_read_ ## pgm_type(&table[index]); \
293  }
